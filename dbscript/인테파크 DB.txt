--정지 사유
DROP TABLE loginstop CASCADE CONSTRAINTS;
DROP SEQUENCE LOGINSTOP_SEQ;
CREATE SEQUENCE "INTEPARK"."LOGINSTOP_SEQ"  MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 1 NOCACHE  NOORDER  NOCYCLE ;

CREATE TABLE loginstop (
  stopno number primary key,
   stoptarget varchar2(500) null,
   stopcause VARCHAR2(100) NULL,
   stopstartdate DATE DEFAULT SYSDATE NOT NULL,
   stopfinishdate DATE DEFAULT SYSDATE NOT NULL,
   stopterm varchar2(10) NOT NULL,
   stopremove varchar2(500) null,
   logid VARCHAR2(15) REFERENCES logininfo(logid) on delete cascade
);

COMMENT ON COLUMN loginstop.stopno IS '정지번호';
COMMENT ON COLUMN loginstop.stoptarget IS '정지위치';
COMMENT ON COLUMN loginstop.stopcause IS '정지사유';
COMMENT ON COLUMN loginstop.stopstartdate IS '정지시작날짜';
COMMENT ON COLUMN loginstop.stopfinishdate IS '정지끝날짜';
COMMENT ON COLUMN loginstop.stopterm IS '정지기한';
COMMENT ON COLUMN loginstop.stopremove IS '정지해제';
COMMENT ON COLUMN loginstop.logid IS '아이디';

drop trigger trigger_logstop;
CREATE OR REPLACE TRIGGER trigger_logstop
AFTER INSERT ON loginstop
FOR EACH ROW
BEGIN
  update logininfo set ENABLED=0, MEMBERSTATE='정지' where logid = :NEW.logid;
END;
/

-- 1. 시공사 테이블에 새로운 CLOB 컬럼을 추가
-- ALTER TABLE [테이블명] ADD ([임시컬럼명] CLOB);
ALTER TABLE ALBUM ADD (ALBUM_INTRO_CLOB CLOB);
-- 2. 기존의 컬럼을 새로운 CLOB 컬럼에 복사한다.
-- UPDATE [테이블명] SET [임시컬럼명] = [기존컬럼명];
-- [임시컬럼] 에 [기존컬럼]에 있던 값을 복사
UPDATE ALBUM SET ALBUM_INTRO_CLOB = ALBUMCONTENTS;
-- 3. [기존컬럼]을 삭제
--ALTER TABLE [테이블명] DROP COLUMN [기존컬럼명];
ALTER TABLE ALBUM DROP COLUMN ALBUMCONTENTS;
-- 4. [임시컬럼명]을 [기존컬럼명]으로 바꾼다.
ALTER TABLE ALBUM RENAME COLUMN ALBUM_INTRO_CLOB TO ALBUMCONTENTS;
